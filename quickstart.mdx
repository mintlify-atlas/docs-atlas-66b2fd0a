## Create Your First Client

Start by creating a VirtualizorJS client with your server credentials:

```typescript
import { createVirtualizorClient } from 'virtualizorjs';

const client = createVirtualizorClient({
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
});
```

<Note>
  The client defaults to HTTPS on port 4085 with `rejectUnauthorized: false` to handle self-signed certificates. See [Configuration Options](#configuration-options) below to customize these settings.
</Note>

## List VPS Instances

Retrieve all VPS instances managed by your Virtualizor server:

```typescript
import { createVirtualizorClient, VirtualizorApiError } from 'virtualizorjs';

const client = createVirtualizorClient({
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
});

try {
  const vpsList = await client.vps.list();

  console.log('VPS Instances:');
  for (const [vpsId, vps] of Object.entries(vpsList)) {
    console.log(`  ${vpsId}: ${vps.hostname}`);
    console.log(`    OS: ${vps.os_name}`);
    console.log(`    RAM: ${vps.ram}`);
    console.log(`    Bandwidth: ${vps.bandwidth}`);
  }
} catch (error) {
  if (error instanceof VirtualizorApiError) {
    console.error(`API Error [${error.code}]: ${error.message}`);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

## Start a VPS

Most VPS operations are asynchronous and return a task ID. Use `client.tasks.wait()` to poll until the operation completes:

```typescript
import { createVirtualizorClient, VirtualizorApiError } from 'virtualizorjs';

const client = createVirtualizorClient({
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
});

try {
  console.log('Starting VPS...');
  const result = await client.vps.start('123');

  if (result.taskid) {
    console.log(`Task ID: ${result.taskid}`);
    console.log('Waiting for task completion...');

    const task = await client.tasks.wait(result.taskid, {
      pollIntervalMs: 2000,  // Check every 2 seconds (default)
      timeoutMs: 60000,      // Timeout after 60 seconds (default: 120000)
    });

    console.log('VPS started successfully!');
    console.log(`Task status: ${task.status}`);
  }
} catch (error) {
  if (error instanceof VirtualizorApiError) {
    console.error(`API Error [${error.code}]: ${error.message}`);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

<Tip>
  Task polling automatically retries at the specified interval until the task completes or the timeout is reached. This eliminates the need for manual polling loops.
</Tip>

## Create a VPS

Create a new VPS instance with basic configuration:

```typescript
import { createVirtualizorClient, VirtualizorApiError } from 'virtualizorjs';
import type { CreateVPSParams } from 'virtualizorjs';

const client = createVirtualizorClient({
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
});

const params: CreateVPSParams = {
  hostname: 'my-new-vps.example.com',
  rootpass: 'SecurePassword123!',
  osid: 1,                  // OS template ID
  ram: 2048,                // RAM in MB
  space: 20,                // Disk space in GB
  bandwidth: 1000,          // Bandwidth in GB
  cpu: 2,                   // CPU cores
};

try {
  console.log('Creating VPS...');
  const result = await client.vps.create(params);

  if (result.taskid) {
    console.log(`VPS creation task initiated: ${result.taskid}`);
    console.log('Waiting for task completion...');

    const task = await client.tasks.wait(result.taskid, {
      pollIntervalMs: 5000,   // Check every 5 seconds
      timeoutMs: 300000,      // Timeout after 5 minutes
    });

    console.log('VPS created successfully!');
  }
} catch (error) {
  if (error instanceof VirtualizorApiError) {
    console.error(`API Error [${error.code}]: ${error.message}`);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

## Stop and Restart Operations

Perform other common VPS lifecycle operations:

<CodeGroup>

```typescript Stop VPS
const result = await client.vps.stop('123');
await client.tasks.wait(result.taskid!);
console.log('VPS stopped');
```

```typescript Restart VPS

await client.tasks.wait(result.taskid!);

```



## Configuration Options

Customize the client with additional configuration options:

```typescript
import { createVirtualizorClient } from 'virtualizorjs';
import type { VirtualizorConfig } from 'virtualizorjs';

const config: VirtualizorConfig = {
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
  
  // Optional settings:
  port: 4085,                    // Default: 4085 (4084 for HTTP)
  https: true,                   // Default: true
  rejectUnauthorized: false,     // Default: false (allows self-signed certs)
  timeout: 30000,                // Default: 30000ms (30 seconds)
};

const client = createVirtualizorClient(config);
```

<Warning>
  **Self-Signed SSL Certificates**: Virtualizor typically uses self-signed SSL certificates. The SDK defaults to `rejectUnauthorized: false` to handle this. Only set `rejectUnauthorized: true` if your Virtualizor instance has a valid, CA-signed certificate.
</Warning>

## Error Handling

All SDK errors are instances of `VirtualizorApiError`. Always catch and handle them appropriately:

```typescript
import { VirtualizorApiError } from 'virtualizorjs';

try {
  await client.vps.start('invalid-id');
} catch (error) {
  if (error instanceof VirtualizorApiError) {
    // API-specific error with code and message
    console.error(`API Error [${error.code}]: ${error.message}`);
    
    // Handle specific error codes
    if (error.code === 404) {
      console.error('VPS not found');
    }
  } else if (error instanceof Error && error.message.includes('timeout')) {
    // Task polling timeout
    console.error('Operation timed out');
  } else {
    // Unexpected error
    console.error('Unexpected error:', error);
  }
}
```

## Complete Example

Here's a complete, production-ready example that combines everything:

```typescript
import { createVirtualizorClient, VirtualizorApiError } from 'virtualizorjs';
import type { CreateVPSParams } from 'virtualizorjs';

// Initialize client
const client = createVirtualizorClient({
  host: process.env.VIRTUALIZOR_HOST!,
  apiKey: process.env.VIRTUALIZOR_API_KEY!,
  apiPass: process.env.VIRTUALIZOR_API_PASS!,
});

async function main() {
  try {
    // List existing VPS
    console.log('Fetching VPS list...');
    const vpsList = await client.vps.list();
    console.log(`Found ${Object.keys(vpsList).length} VPS instances\n`);

    // Create a new VPS
    const params: CreateVPSParams = {
      hostname: 'demo-vps.example.com',
      rootpass: 'SecurePassword123!',
      osid: 1,
      ram: 2048,
      space: 20,
      bandwidth: 1000,
      cpu: 2,
    };

    console.log('Creating new VPS...');
    const createResult = await client.vps.create(params);

    if (createResult.taskid) {
      console.log(`Task ID: ${createResult.taskid}`);
      console.log('Waiting for creation to complete...');

      const task = await client.tasks.wait(createResult.taskid, {
        pollIntervalMs: 5000,
        timeoutMs: 300000,
      });

      console.log('VPS created successfully!');
      console.log(`Task status: ${task.status}`);
    }

    // List VPS again to see the new one
    const updatedList = await client.vps.list();
    console.log(`\nTotal VPS instances: ${Object.keys(updatedList).length}`);
  } catch (error) {
    if (error instanceof VirtualizorApiError) {
      console.error(`API Error [${error.code}]: ${error.message}`);
      process.exit(1);
    } else {
      console.error('Unexpected error:', error);
      process.exit(1);
    }
  }
}

main();
```
